ཀྲུང་ཁེ་ཧྭ་ཏ་རྒྱལ་སྤྱིའི་ཚོང་ལས་ཀྱི་ཚད་བཀག་འབྲེལ་ཡོད་ཚོང་ལས། (གུང་དུ “ཀྲུང་ཁེ་ཧྭ་ཏ” ཡང་ན “ང་ཚོ”) ཁྱེད་རང་གི་སྒེར་གྱི་གསང་དོན་ལ་བརྩི་བཀུར་བྱེད་ཅིང་། ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་སྲུང་སྐྱོབ་བྱེད་པར་ཤུགས་ཆེ་ཤོས་ཀྱི་འབད་བརྩོན་བྱེད་རྒྱུ་ཡིན། འདི་ལྟར “EMUI Interconnect གི་སྒེར་གྱི་གསང་དོན་སྲིད་བྱུས” (“སྲིད་བྱུས་འདི”) ནི་ཁྱེད་རང་གིས་EMUI Interconnect བེད་སྤྱོད་ཀྱི་སྒྲིག་ཆས (འབྲེལ་ཡོད་ནང་དོན་ཡོངས་དང་མཉམ་དུ། ཕལ་ཆེར “ཞབས་ཞུ་འདི” ཞེས་འབོད) བེད་སྤྱོད་བྱེད་སྐབས་ཀྱི་བེད་སྤྱོད་པའི་གནས་ཚུལ་བསྡུ་རུབ་དང་། བཀོལ་སྤྱོད། ཉར་འཇོག། བརྒྱུད་སྤྲོད། སྲུང་སྐྱོབ་བཅས་ལ་སྤྱི་ཚུལ་དུ་འཇུག ཞབས་ཞུ་འདི་བེད་སྤྱོད་བྱེད་སྔོན། སྲིད་བྱུས་འདི་ནན་ཏན་གྱིས་ཀློག་ལ་རྟོགས་པ་དང་མོས་མཐུན་ཡོད་པར་གཏན་འབེབས་བྱེད་དགོས། ལྷག་པར་དུ་སྲ་པོའི་ཡི་གེ་ལྟར་མཚོན་པའི་གལ་ཆེའི་ནང་དོན་ལ་དོ་སྣང་བྱེད་རོགས། ཁྱེད་རང་གིས་བེད་སྤྱོད་ཀྱི་སྒྲིག་ཆས་འདི་བེད་སྤྱོད་བྱེད་སྐབས་སྣུམ་ཚད་སླེབས་པར་ངེས་པར་དུ་གཏན་འབེབས་བྱེད་དགོས། གལ་ཏེ་ཁྱེད་རང་ན་གཞོན་ཡིན་ན། སྒྲིག་ཆས་འདི་བེད་སྤྱོད་བྱེད་པར་སྲུང་སྐྱོབ་པའི་མོས་མཐུན་དང་སྲིད་བྱུས་འདི་དང་འབྲེལ་ཡོད་ཞབས་ཞུའི་དོན་ཚན་ངོས་ལེན་བྱེད་དགོས། ཞབས་ཞུ་འདི་བེད་སྤྱོད་བྱེད་པ་ནི་སྲིད་བྱུས་འདིའི་དོན་ཚན་དག་ཁྱེད་རང་གིས་ཀློག་པ་དང་། ངོས་ལེན། མོས་མཐུན་བྱས་པར་བརྩི། ང་ཚོས་ཞབས་ཞུ་སྤྲོད་པའི་དགོས་ངེས་དང་ནུས་པའི་རྒྱབ་ལྗོངས་ལྟར། སྲིད་བྱུས་འདིའི་དགོངས་དོན་ལ་ནན་ཏན་གྱིས་བརྩི་སྲུང་བྱེད་དེ་ཁྱེད་ཀྱི་གནས་ཚུལ་བེད་སྤྱོད་དང་སྲུང་སྐྱོབ་བྱེད་རྒྱུ་ཡིན།

(ཀ) ཁྱེད་ཀྱི་ཡོ་བྱད་ཀྱི་གནས་ཚུལ། ཁྱེད་རང་གིས་ཡོ་བྱད་ངེས་ཅན་ཞིག་གི་སྟེང་ཞབས་ཞུ་འདི་བེད་སྤྱོད་བྱེད་སྐབས། ང་ཚོས་ཡོ་བྱད་དེ་ལས་ངེས་པར་དུ་གྱི་གནས་ཚུལ་བསྡུ་རྒྱུ་ཡིན། དེ་དག་ལ། ཡོ་བྱད་ངོས་འཛིན་ཨང་གྲངས། ཡོ་བྱད་དཔེ་གཞི། ཡོ་བྱད་མཚན་གཟུགས། ལམ་ལུགས་པར་མ། དྲ་བའི་གནས་ཚུལ་ (ས་གནས་ཀྱི་IP ཁ་བྱང་། གཞུང་ལམ་གྱི་IP ཁ་བྱང་) བཅས་ཡིན། དེ་དག་ལ་བརྟེན་ནས་པར་མ་ངོས་འཛིན་དང་། འཛོལ་བ་མྱུར་པོར་ཞིབ་བཤེར། དཀའ་ངལ་སེལ་ཐབས། ཁྱེད་རང་ལ་གཞན་དང་མཐུན་པའི་ཚད་གཞི་སྤྲོད་པར་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོད། དེ་ལྟར་བྱས་ནས་ཚོང་ཟོག་དུས་ཐོག་ཏུ་ལེགས་བཅོས་བྱེད་ཐུབ་པ་དང་། པར་མ་གསར་བརྗེའི་བརྡ་ཐོ་སྤྲོད་ཐུབ།

(ཁ) ཁྱེད་ཀྱི་ལོགས་སུ་བཀོད་པའི་གནས་ཚུལ། ཞབས་ཞུ་འདི་བེད་སྤྱོད་བྱེད་སྐབས། ང་ཚོས་ཞབས་ཞུ་འདིའི་ལོགས་སུ་བཀོད་པའི་གནས་ཚུལ་ (བེད་སྤྱོད་བྱེད་ཚད། བརྒྱལ་སྤྲོད་བྱས་པའི་གནས་ཚུལ་གྱི་ཚད། གནས་ཚུལ་རིགས། བྱ་འགུལ་གྱི་གནས་སྟངས། སྦྲེལ་མཐུད་ཀྱི་གནས་སྟངས་སོགས) བསྡུ་རྒྱུ་ཡིན། ཁྱེད་རང་ལས་བསྡུས་པའི་ལོགས་སུ་བཀོད་པའི་གནས་ཚུལ་དེ་དག་གཞི་རྩའི་ཞབས་ཞུ་སྤྲོད་པ་དང་། ཞབས་ཞུ་བེད་སྤྱོད་བྱེད་སྐབས་བྱུང་བ་དང་། བྱུང་སྲིད་པའི་དཀའ་ངལ་དང་འཛོལ་བ་རྟོག་ཞིབ་བྱེད་པར་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོད། དེ་ནི་ཁྱེད་ཀྱི་མི་སྒེར་གྱི་ངོས་འཛིན་དང་ཡང་ན་འབྲེལ་བ་མེད།

(ང) བེད་སྤྱོད་པའི་བརྒྱུད་སྤྲོད་ཀྱི་གནས་ཚུལ། ཁྱེད་རང་གིས་ཡིག་ཚང་སྤེལ་སྒྲིག་སོགས་ལས་ནུས་བེད་སྤྱོད་བྱེད་སྐབས། ང་ཚོས་ཁྱེད་རང་གིས་གདམ་ག་བཏང་བའི་པར་རིས། བརྙན་ཕབ། ཡི་གེ། འབྲེལ་མཐུད་བྱ་ཡུལ་སོགས་ནང་དོན་དང་། དེ་བཞིན་ཁྱེད་རང་གིས་སྤེལ་སྒྲིག་བྱེད་བཞིན་པའི་བེད་སྤྱོད་ཡང་ན་ཞབས་ཞུའི་སྡེ་ཚན་གྱི་ཆོག་མཆན་གྱི་གནས་སྟངས་བཅས་གསོག་སྟེགས་སུ་བཞག་དགོས། དེ་ལྟར་བྱས་ནས་གོང་གསལ་གནས་ཚུལ་གྱི་ནང་དོན་སྤེལ་སྒྲིག་སྒྲུབ་པར་རོགས་རམ་བྱེད་ཐུབ། དེ་དག་ཡོངས་རྫོགས་སྤེལ་སྒྲིག་སྒྲུབ་པར་བེད་སྤྱོད་བྱེད་ཀྱི་ཡོད། ང་ཚོས་ནང་དོན་ངོས་འཛིན་དང་། དགོངས་པ་ངོས་འཛིན་བྱེད་ཀྱི་མེད། དེ་བཞིན་དམིགས་ཡུལ་གཞན་དུ་བེད་སྤྱོད་མི་བྱེད།

(༢) ཁྱེད་རང་གིས་ལས་ནུས་འདི་བེད་སྤྱོད་བྱས་ཏེ་གཞན་པའི་ཡོ་བྱད་དང་སྦྲེལ་མཐུད་བྱེད་ཐུབ། དེ་ལྟར་བྱས་ནས་གཞན་པའི་ཡོ་བྱད་དེའི་ཐོག་ནས་འབྲེལ་མཐུད་བྱ་ཡུལ། ཡིག་ཚང་། པར་རིས། བརྙན་ཕབ་སོགས་རྒྱབ་སྐྱོར་བྱེད་པའི་ནང་དོན་ཐོབ་ཐུབ། དེ་ལྟར་སྒྲུབ་པར། ང་ཚོས་ཁྱེད་རང་གིས་གདམ་ག་བཏང་བའི་གོང་གསལ་བེད་སྤྱོད་པའི་ནང་དོན་ཀློག་དགོས། གོང་གསལ་ནང་དོན་དེ་དག་ནི་ཁྱེད་རང་གིས་ཡིག་ཚང་ཕན་ཚུན་སྤེལ་སྒྲིག་གི་ལས་ནུས་བེད་སྤྱོད་བྱེད་སྐབས་ཁྱེད་ཀྱི་མོས་མཐུན་བྱུང་རྗེས་ང་ཚོས་རང་འགུལ་གྱིས་བསྡུ་ཞིང་བཀོལ་སྤྱོད་བྱེད་རྒྱུ་ཡིན།

    ༡. མོས་མཐུན་ཐོབ་སྐབས་ཀྱི་བརྒྱུད་སྤྲོད། ཁྱེད་རང་གི་མོས་མཐུན་ཐོབ་པའི་རྗེས། ཀྲུང་ཁེ་ཧྭ་ཏས་ཁྱེད་རང་གི་ཆོག་མཆན་གྱི་ས་མཚམས་ནང་དུ་ཁྱེད་རང་གིས་གསལ་པོར་གཏན་འབེབས་བྱས་པའི་གསུམ་པའི་འགོ་ཁྱབ་ལ་གནས་ཚུལ་བརྒྱུད་སྤྲོད་བྱེད་རྒྱུ་ཡིན།

ཀྲུང་ཁེ་ཧྭ་ཏ་ནི་རྒྱལ་ཡོངས་རྒྱུགས་མེད་ཀྱི་ཚོང་ལས་ཤིག་ཡིན། དེ་ལྟར་བྱས་ནས་ཀྲུང་ཁེ་ཧྭ་ཏས་བསྡུ་རུབ་བྱས་པའི་ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་དེ་དག་ཁྱེད་རང་གིས་ཚོང་ཟོག་ཡང་ན་ཞབས་ཞུ་བེད་སྤྱོད་བྱེད་པའི་རྒྱལ་ཁབ་ ཡང་ན་ས་ཁུལ། ཡང་ན་ཀྲུང་ཁེ་ཧྭ་ཏ་ ཡང་ན་དེའི་འབྲེལ་ཡོད་ཚོང་ལས་དང་། ཡན་ལག་ཚོང་ལས། ཞབས་ཞུའི་མཁོ་སྤྲོད་པ། ཚོང་ལས་མཉམ་ལས་པ་བཅས་ལ་ལས་ཁུངས་ཡོད་པའི་རྒྱལ་ཁབ་གཞན་པར་བཀོལ་སྤྱོད་བྱེད་སྲིད། ཡང་ན་རྒྱལ་ཁབ་དེ་དག་ནས་ཐེག་ཆོག རྒྱལ་ཁབ་དེ་དག་གི་གནས་ཚུལ་སྲུང་སྐྱོབ་ཀྱི་ཁྲིམས་ལུགས་མི་འདྲ་བ་ཡོད་སྲིད། དེ་ལྟ་བུའི་གནད་དོན་དག་ལ། ཀྲུང་ཁེ་ཧྭ་ཏས་ང་ཚོས་བསྡུ་རུབ་བྱས་པའི་གནས་ཚུལ་དེ་དག་འདི་ལྟར་གསལ་བསྒྲགས་དང་སྒྲུབ་སྟངས་ཀྱི་ཁྲིམས་ལུགས་ཀྱི་དགོངས་དོན་ལྟར་བཀོལ་སྤྱོད་བྱེད་པར་འགན་ལེན་བྱེད་པའི་བྱ་ཐབས་ལག་བསྟར་བྱེད་རྒྱུ་ཡིན།

ང་ཚོས་རྒྱལ་ཁབ་ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ས་ཁོངས་ནང་དུ་བསྡུ་རུབ་དང་བཀོལ་སྤྱོད་བྱས་པའི་སྒེར་གྱི་གནས་ཚུལ་དེ་དག་རྒྱལ་ཁབ་ཀྲུང་ཧྭ་མི་དམངས་སྤྱི་མཐུན་རྒྱལ་ཁབ་ཀྱི་ས་ཁོངས་ནང་ཉར་འཇོག་བྱེད་རྒྱུ་ཡིན། གལ་ཏེ་དམིགས་བསལ་གྱི་ཚོང་ཟོག་ཡང་ན་ཞབས་ཞུ་དེར་སྒེར་གྱི་གནས་ཚུལ་རྒྱལ་ཡོངས་བརྒལ་བར་དགོས་པའི་གནད་དོན་ཡོད་ན། ང་ཚོས་ཁྲིམས་ལུགས་ཀྱི་དགོངས་དོན་སྒྲུབ་པའི་རྗེས (དཔེར་ན། ལྟ་སྐུལ་ལས་ཁུངས་ཀྱི་དགོངས་དགོས་ཀྱི་བདེ་འཇགས་དཔྱད་བསྡུར་བརྒྱུད་པ་ལྟར) སྒེར་གྱི་གནས་ཚུལ་དེ་དག་རྒྱལ་ཡོངས་ཀྱི་རྙེད་མཁན་ལ་སྤྲོད་རྒྱུ་ཡིན།

༧. སྒེར་གྱི་གནས་ཚུལ་ཉར་འཇོག་དང་སྲུང་སྐྱོབ།

(༡) ཉར་འཇོག་གི་དུས་ཡུན་དང་དུས་ཡུན་ལས་བརྒལ་བའི་བཀོལ་སྤྱོད། ཁྲིམས་ལུགས་དང་སྒྲིག་བྱ་གཞན་དུ་གཏན་འབེབས་མེད་ན། ང་ཚོས་སྲིད་བྱུས་འདིར་གསལ་པོར་བཀོད་པའི་དམིགས་ཡུལ་སྒྲུབ་པར་དགོས་པའི་དུས་ཡུན་ཐུང་ངུ་ཡོངས་རྫོགས་ཀྱི་ནང་དུ་ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་ཉར་འཇོག་བྱེད་རྒྱུ་ཡིན། ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་ང་ཚོས་ཉར་འཇོག་བྱེད་པའི་དུས་ཡུན་དང་བརྒལ་སྐབས། ང་ཚོས་ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་དེ་དག་བཟོ་བཅོས་དང་། ཡང་ན་མིང་གཞན་དུ་བསྒྱུར་རྒྱུ་ཡིན།

(༢) གནས་ཚུལ་གྱི་བདེ་འཇགས་དང་སྲུང་སྐྱོབ་ཀྱི་བྱ་ཐབས། ང་ཚོས་བདེ་འཇགས་ཞིབ་བཤེར་དང་། གསང་རྟགས་ལག་ཆ་དང་སྒྲིག་ཆས་བེད་སྤྱོད། གཞན་པའི་རྒྱུ་མཚན་ལྡན་པའི་བདེ་འཇགས་ཀྱི་བྱ་ཐབས་དང་ལམ་ལུགས་སོགས་ལ་བརྟེན་ནས་ཁྱེད་ཀྱི་སྒེར་གྱི་གནས་ཚུལ་སྲུང་སྐྱོབ་བྱེད་རྒྱུ་ཡིན། ང་ཚོས་ཞབས་ཞུ་འདིའི་ལས་དོན་སྡོད་ལྕོག་བྱེད་པའམ་ཡང་ན་ཁྱེད་རང་གིས་ཞབས་ཞུ་འདི་བེད་སྤྱོད་བྱེད་པར་སྤོང་སྐབས། ང་ཚོས་ཁྱེད་ཀྱི་གནས་ཚུལ་གང་ཡང་རུང་བ་བསྡུ་རུབ་དང་བེད་སྤྱོད་བྱེད་མི་ཆོག དེ་བཞིན་ཉར་འཇོག་བྱས་པའི་གནས་ཚུལ་དེ་དག་མིང་གཞན་དུ་བསྒྱུར་རམ་བཟོ་བཅོས་བྱེད་རྒྱུ་ཡིན།

༨. སྒེར་གྱི་གནས་ཚུལ་དོ་དམ།
